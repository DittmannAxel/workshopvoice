AGENT_TOKEN=$(az account get-access-token --resource "https://ai.azure.com" --query accessToken -o tsv)

# Option A (classic assistants): returns OpenAI-style assistant IDs like `asst_...`
curl --request GET \
  --url "https://adminfeb02-resource.services.ai.azure.com/api/projects/adminfeb02/assistants?api-version=v1" \
  --header "Authorization: Bearer $AGENT_TOKEN"

# Option B (versioned Foundry agents): returns agents with name/version.
# Some VoiceLive setups expect `agentName:version` (example: `agentFeborder:8`).
curl --request GET \
  --url "https://adminfeb02-resource.services.ai.azure.com/api/projects/adminfeb02/agents?api-version=2025-11-15-preview" \
  --header "Authorization: Bearer $AGENT_TOKEN"
